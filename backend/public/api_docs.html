SellMe

Routes and Required Params

USERS

GET: / users#index POST: /users/login users#login
  • required params: {:username or :email, :password}
GET /users/:user_id users#show
  • required params: {:user_id}
POST: /users users#create
  • required params: {:username, :password, :name, :email, :avatar, :bio}
GET /users/:user_id/listings users#listings
  • required params: {:user_id}
GET: /users/:user_id/offers users#offers
  • required params: {:user_id}
DELETE: /users/:user_id users#destroy
  • required params: {:user_id}

LISTINGS

GET: /listings listings#index POST: /listings listings#create
  • required params: {:user_id, :name, :description, :image_url, :category_id, :max_price, :accept_price, :expires_at}
GET: /listings/:id listings#show
  • required params: {:id}
PUT: /listings/:id listings#update
  • required params: {:id}
DELETE: /listings/:id listings#destroy
  • required params: {:id}

OFFERS

  • POST: /listings/:listing_id/offers offers#create
    • required params: {:user_id, :name, :description, :image_url, :price, :listing_id}
    POST: /offers/invalidate offers#invalidate
    • required params: {:offer_id}
    POST: /offers/accept offers#accepts
    • required params: {:offer_id}
    DELETE /listings/:listing_id/offers/:id offers#destroy
    • required params: {:offer_id}
